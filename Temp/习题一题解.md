#include <bits/stdc++.h>

## 题目K_1004

### ACM解法

```c++
#include <bits/stdc++.h>
using namespace std;
int main( ) {
    int Cases;
    cin >> Cases;
    while (Cases--) {
        int Length;
        cin >> Length;
        double Num, MaxSum = INT_MIN, MinSum = INT_MAX, Sum = 0;
        for (int i = 0; i < Length; ++i) {
            cin >> Num;
            MaxSum = max(Num, MaxSum);
            MinSum = min(Num, MinSum);
            Sum += Num;
        }
        printf("%.2f %.2f %.2f\n", MaxSum, MinSum, Sum / Length);
    }
    return 0;
}

```



## 题目M_1007

### ACM解法

```c++
#include <bits/stdc++.h>
using namespace std;
int main() {
    int n;
    cin >> n;
    while (n--) {
        int Num, NumPow, NumberDigits, NumSum = 1;
        cin >> Num >> NumPow >> NumberDigits;
        NumberDigits = (int)pow((double)10, (double)NumberDigits);
        for (int i = 0; i < NumPow; ++i)
            NumSum *= Num, NumSum %= NumberDigits;
        cout << NumSum << endl;
    }
    return 0;
}
```
